Smart Contract Architecture

A single C++ EOSIO contract handles all game logic. Trustless, auditable, and immutable once deployed.

▼ Contract Actions

mintchar()
Mint a new NFT character — class, race, name, and initial stats generated on-chain
battle()
Initiate PvP combat; winner determined by d20 rolls against AC, all resolved on-chain
gainexp()
Award XP to a character, triggering level-up logic and ASI at thresholds
equip()
Equip an item NFT to a character slot, updating combat stats accordingly
rest()
Long rest restores HP to max and resets short-rest resources like hit dice

// D&D 5e Character Classes
enum character_class : uint8_t {
    BARBARIAN = 0,  BARD     = 1,
    CLERIC    = 2,  DRUID    = 3,
    FIGHTER   = 4,  MONK     = 5,
    PALADIN   = 6,  RANGER   = 7,
    ROGUE     = 8,  SORCERER = 9,
    WARLOCK   = 10, WIZARD   = 11
};

// Character NFT with full D&D 5e stats
TABLE character {
    uint64_t     id;
    name         owner;
    string       char_name;
    uint8_t      char_class;
    uint8_t      char_race;
    uint8_t      rarity_level;
    uint8_t      level;
    uint32_t     experience;
    stats        ability_scores;
                 // STR DEX CON INT WIS CHA
    combat_stats combat;
                 // HP, AC, proficiency
    uint32_t     wins;
    uint32_t     losses;

    uint64_t primary_key() const {
        return id;
    }
    uint64_t by_owner() const {
        return owner.value();
    }
    uint64_t by_level() const {
        return level;
    }

    EOSLIB_SERIALIZE(character,
        (id)(owner)(char_name)
        (char_class)(char_race)
        (rarity_level)(level)
        (experience)(ability_scores)
        (combat)(wins)(losses)
    )
};
